Usage

Send a post request to http://pacificpygmyowl.herokuapp.com/api/query

AngularJS example (make sure to include $http in the controller)

$http.post('http://pacificpygmyowl.herokuapp.com/api/query', input, {headers: {'Content-Type': 'application/json'} })
	.then()(function (response) {
		//do something
	},
	function(err) {
		//handle error
	});

NodeJS example (Requires request module)

var request = require('request');

var options = {
	url: 'http://pacificpygmyowl.herokuapp.com/api/query',
	method: 'POST',
	json: {
		"start_date": "yyyy-MM-ddTHH:mm:ss.SSSZ",
		"end_date": "yyyy-MM-ddTHH:mm:ss.SSSZ",
		"tpc_list": "[]",
		"instr_list": "[]"
	}
};

request(options, function (error, response, body) {
	if (!error && response.statusCode == 200) {
		console.log(body.id)
	}
});
